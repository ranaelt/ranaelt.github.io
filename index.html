SIGPROC Header Field Creation Functions



                void CreateHeaderFieldTelescopeId(int value)               // creates header field *telescope_id*: 0=fake data; 1=Arecibo; 2=Ooty... others to be added
                void CreateHeaderFieldMachineId(int value)                 // creates header field *machine_id*: 0=FAKE; 1=PSPM; 2=WAPP; 3=OOTY... others to be added
                void CreateHeaderFieldDataType(int value)                  // creates header field *data_type*: 1=filterbank; 2=time series... others to be added
                void CreateHeaderFieldRawDataFile(char [] value)           // creates header field *rawdatafile*: the name of the original data file
                void CreateHeaderFieldSourceName(char [] value)            // creates header field *source_name*: the name of the source being observed by the telescope
                void CreateHeaderFieldBaryCentric(int value)               // creates header field *barycentric*: equals 1 if data are barycentric or 0 otherwise
                void CreateHeaderFieldPulsarCentric(int value)             // creates header field *pulsarcentric*: equals 1 if data are pulsarcentric or 0 otherwise
                void CreateHeaderFieldAzStart(double value)                // creates header field *az_start*: telescope azimuth at start of scan (degrees)
                void CreateHeaderFieldZaStart(double value)                // creates header field *za_start*: telescope zenith angle at start of scan (degrees)
                void CreateHeaderFieldSrcRaj(double value)                 // creates header field *src_raj*: right ascension (J2000) of source (hhmmss.s)
                void CreateHeaderFieldSrcDej(double value)                 // creates header field *src_dej*: declination (J2000) of source (ddmmss.s)
                void CreateHeaderFieldTStart(double value)                 // creates header field *tstart*: time stamp (MJD) of first sample
                void CreateHeaderFieldTSamp(double value)                  // creates header field *tsamp*: time interval between samples (s)
                void CreateHeaderFieldNBits(int value)                     // creates header field *nbits*: number of bits per time sample
                void CreateHeaderFieldNSamples(int value)                  // creates header field *nsamples*: number of time samples in the data file (rarely used any more)
                void CreateHeaderFieldFch1(double value)                   // creates header field *fch1*: centre frequency (MHz) of first filterbank channel
                void CreateHeaderFieldFoff(double value)                   // creates header field *foff*: filterbank channel bandwidth (MHz)
                void CreateHeaderFieldFrequencyStart(character value)      // creates header field *frequency_start*: start of frequency table (see below for explanation)
                void CreateHeaderFieldFChannel(double value)               // creates header field *fchannel*: frequency channel value (MHz)
                void CreateHeaderFieldFrequencyEnd(character value)        // creates header field *frequency_end*: end of frequency table (see below for explanation)
                void CreateHeaderFieldNChans(int value)                    // creates header field *nchans*: number of filterbank channels
                void CreateHeaderFieldNIfs(int value)                      // creates header field *nifs*: number of seperate IF channels
                void CreateHeaderFieldReFdm(double value)                  // creates header field *refdm*: reference dispersion measure (cm-3 pc)
                void CreateHeaderFieldPeriod(double value)                 // creates header field *period*: folding period (s)
